   RESPOSTA : LITERAL
   CONTINUA : LÓGICO

   CONTINUA <--- VERDADE
   ESCREVA 'ENTRE UM NUMERO'
  
   ENQUANTO CONTINUA
     LEIA NUM
     SOMA <--- 0
     DIVISOR <--- NUM

     SE NUM > 0

       ENQUANTO DIVIDENDO > 0 FAÇA

         RESTO <---- NUM MODE DIVIDENDO
         SE RESTO = 0
           SOMA <---- SOMA + 1 
           QUOCIENTE <--- NUM DIV DIVIDENDO
           ESCREVA QUOCIENTE + ' É DIVISOR DE ' NUM
           DIVIDENDO <---- DIVIDENDO - 1
         FIM-SE
        
       FIM-ENQUANTO       

       SE SOMA <= 2
            ESCREVA  NUM + ' É NÚMERO PRIMO'
       FIM-SE
     
       ESCREVA 'DESEJA ENTRAR OUTRO NUMERO? ( SIM/NAO )'
       LEIA RESPOSTA
       SE RESPOSTA = 'NAO'
          CONTINUA <--- FALSO
       FIM-SE
  
       
     SENAO
	ESCREVA ' ENTRE O NUMERO > 0 '
     FIM-SE   
 
   FIM-CONTINUA

FIM-INICIO
